State Management

The app uses SolidJS signals for reactive state management:

interface User {
  id: string;
  username: string;
  email: string;
  avatar?: string;
  level: number;
  experiencePoints: number;
  achievements: Achievement[];
}

interface Notebook {
  id: string;
  title: string;
  description?: string;
  color: string;
  isPublic: boolean;
  createdAt: Date;
  updatedAt: Date;
  pageCount: number;
}

interface Note {
  id: string;
  notebookId: string;
  title: string;
  content: string;
  contentType: "markdown" | "rich-text" | "code";
  isFavorite: boolean;
  createdAt: Date;
  updatedAt: Date;
  collaborators?: User[];
}
