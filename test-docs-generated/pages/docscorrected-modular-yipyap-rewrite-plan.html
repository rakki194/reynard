// packages/caption-ui/src/components/CaptionGenerator.tsx
export const CaptionGenerator: Component<CaptionGeneratorProps> = (props) => {
  const [selectedModel, setSelectedModel] = createSignal<string>("jtp2");
  const [imageFile, setImageFile] = createSignal<File | null>(null);
  const [isGenerating, setIsGenerating] = createSignal(false);
  const [result, setResult] = createSignal<CaptionResult | null>(null);
  
  const manager = useBackendAnnotationManager();
  
  const generateCaption = async () => {
    if (!imageFile()) return;
    
    setIsGenerating(true);
    try {
      const task: CaptionTask = {
        imagePath: imageFile()!.name,
        generatorName: selectedModel(),
        config: { threshold: 0.2 },
        postProcess: true
      };
      
      const result = await manager.generateCaption(task);
      setResult(result);
    } finally {
      setIsGenerating(false);
    }
  };
  
  return (
    <div class="caption-generator">
      <ModelSelector 
        selected={selectedModel()} 
        onSelect={setSelectedModel}
        available={manager.getAvailableGenerators()}
      />
      <ImageUpload 
        file={imageFile()} 
        onFileChange={setImageFile}
        accept="image/*"
      />
      <GenerateButton 
        onClick={generateCaption}
        disabled={!imageFile() || isGenerating()}
        loading={isGenerating()}
      />
      <Show when={result()}>
        <CaptionResult result={result()!} />
      </Show>
    </div>
  );
};

Deliverables:

  • Interactive caption generation interface

// packages/caption-ui/src/components/ModelManager.tsx
export const ModelManager: Component<ModelManagerProps> = (props) => {
  const [models, setModels] = createSignal<GeneratorInfo[]>([]);
  const [loadedModels, setLoadedModels] = createSignal<Set<string>>(new Set());
  const [healthStatus, setHealthStatus] = createSignal<HealthStatus | null>(null);
  
  const manager = useBackendAnnotationManager();
  
  const loadModel = async (modelName: string) => {
    try {
      await manager.loadModel(modelName);
      setLoadedModels(prev => new Set([...prev, modelName]));
    } catch (error) {
      console.error("Failed to load model:", error);
    }
  };
  
  const unloadModel = async (modelName: string) => {
    try {
      await manager.unloadModel(modelName);
      setLoadedModels(prev => {
        const newSet = new Set(prev);
        newSet.delete(modelName);
        return newSet;
      });
    } catch (error) {
      console.error("Failed to unload model:", error);
    }
  };
  
  return (
    <div class="model-manager">
      <div class="models-grid">
        <For each={models()}>
          {(model) => (
            <ModelCard 
              model={model}
              isLoaded={loadedModels().has(model.name)}
              onLoad={() => loadModel(model.name)}
              onUnload={() => unloadModel(model.name)}
            />
          )}
        </For>
      </div>
      <HealthStatus health={healthStatus()} />
    </div>
  );
};

Deliverables:

  • Model status dashboard

// packages/caption-ui/src/components/BatchProcessor.tsx
export const BatchProcessor: Component<BatchProcessorProps> = (props) => {
  const [files, setFiles] = createSignal<File[]>([]);
  const [selectedModel, setSelectedModel] = createSignal<string>("jtp2");
  const [progress, setProgress] = createSignal<BatchProgress | null>(null);
  const [results, setResults] = createSignal<CaptionResult[]>([]);
  
  const manager = useBackendAnnotationManager();
  
  const processBatch = async () => {
    const tasks: CaptionTask[] = files().map(file => ({
      imagePath: file.name,
      generatorName: selectedModel(),
      config: { threshold: 0.2 },
      postProcess: true
    }));
    
    setProgress({ completed: 0, total: tasks.length, percentage: 0 });
    
    try {
      const results = await manager.generateBatchCaptions(tasks, (progress) => {
        setProgress(progress);
      });
      setResults(results);
    } catch (error) {
      console.error("Batch processing failed:", error);
    }
  };
  
  return (
    <div class="batch-processor">
      <FileUploadZone 
        files={files()}
        onFilesChange={setFiles}
        accept="image/*"
        multiple
      />
      <ModelSelector 
        selected={selectedModel()}
        onSelect={setSelectedModel}
      />
      <Show when={progress()}>
        <ProgressBar progress={progress()!} />
      </Show>
      <BatchResults results={results()} />
    </div>
  );
};

Deliverables:

  • Batch upload interface
  • Progress tracking
  • Result management

// packages/gallery/src/components/MultiModalGrid.tsx
export const MultiModalGrid: Component<MultiModalGridProps> = (props) => {
  const [items, setItems] = createSignal<(ImageItem | AudioItem | VideoItem | TextItem)[]>([]);
  const [searchQuery, setSearchQuery] = createSignal("");
  const [selectedItems, setSelectedItems] = createSignal<Set<string>>(new Set());
  
  const filteredItems = createMemo(() => {
    const query = searchQuery().toLowerCase();
    if (!query) return items();
    
    return items().filter(item => 
      item.name.toLowerCase().includes(query) ||
      item.metadata?.tags?.some(tag => tag.toLowerCase().includes(query))
    );
  });
  
  return (
    <div class="multi-modal-grid">
      <SemanticSearch 
        query={searchQuery()}
        onQueryChange={setSearchQuery}
        onResults={setItems}
      />
      <div class="grid-container">
        <For each={filteredItems()}>
          {(item) => (
            <Show 
              when={item.type === "image"}
              fallback={
                <Show 
                  when={item.type === "audio"}
                  fallback={
                    <Show 
                      when={item.type === "video"}
                      fallback={<TextItem item={item as TextItem} />}
                    >
                      <VideoItem item={item as VideoItem} />
                    </Show>
                  }
                >
                  <AudioItem item={item as AudioItem} />
                </Show>
              }
            >
              <ImageItem item={item as ImageItem} />
            </Show>
          )}
        </For>
      </div>
    </div>
  );
};

// packages/gallery/src/components/DatasetCleaningModal.tsx
export const DatasetCleaningModal: Component<DatasetCleaningProps> = (props) => {
  const [duplicates, setDuplicates] = createSignal<DuplicateGroup[]>([]);
  const [selectedDuplicates, setSelectedDuplicates] = createSignal<Set<string>>(new Set());
  const [isScanning, setIsScanning] = createSignal(false);
  
  const scanForDuplicates = async () => {
    setIsScanning(true);
    try {
      const duplicates = await props.onScanDuplicates();
      setDuplicates(duplicates);
    } finally {
      setIsScanning(false);
    }
  };
  
  const removeDuplicates = async () => {
    const toRemove = Array.from(selectedDuplicates());
    await props.onRemoveDuplicates(toRemove);
    setDuplicates(prev => prev.filter(group => 
      !group.items.some(item => toRemove.includes(item.id))
    ));
  };
  
  return (
    <div class="dataset-cleaning-modal">
      <div class="scan-section">
        <Button 
          onClick={scanForDuplicates}
          loading={isScanning()}
          disabled={isScanning()}
        >
          Scan for Duplicates
        </Button>
      </div>
      
      <div class="duplicates-section">
        <For each={duplicates()}>
          {(group) => (
            <DuplicateGroup 
              group={group}
              selected={selectedDuplicates()}
              onSelectionChange={setSelectedDuplicates}
            />
          )}
        </For>
      </div>
      
      <div class="actions-section">
        <Button 
          onClick={removeDuplicates}
          disabled={selectedDuplicates().size === 0}
          variant="danger"
        >
          Remove Selected ({selectedDuplicates().size})
        </Button>
      </div>
    </div>
  );
};
