Pseudo class selectors

CSS Modules also support adding pseudo class selectors:

/* component/text.css */
.text {
  color: #777;
  font-weight: 24px;
}

.text:hover {
  color: #f60;
}
/* component/text.js */
import styles from "./text.css";
import React from "react";

export function Text() {
  return <p className={styles.text}>Text with hover</p>;
}
